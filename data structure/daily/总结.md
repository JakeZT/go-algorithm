自定义排序

1.二维数组

```go
func arrSort(pq [][]int) {
	sort.SliceStable(pq, func(a, b int) bool {
		if pq[a][0] == pq[b][0] {
			return pq[a][1] < pq[b][1]
		} else {
			return pq[b][0] <= pq[b][0]
		}
	})
}
```

2.一维数组

```go
func arrSort(pq []int) {
	sort.SliceStable(pq, func(a, b int) bool {
		return pq[a] <= pq[b]
	})
}
```

	m := 0
	for _, i := range a {
		if i > m {
			m = i
		}
	}
	return m
}

func Min(a []int) int {
	m := 0
	for _, i := range a {
		if i < m {
			m = i
		}

自定义任意组合的 map 和 Arr

```go

type myMap map[interface{}]interface{}
type myArr []interface{}

func MapToSlice(input myMap) myArr {
	output := make(myArr, 0)
	for _, val := range input {
		output = append(output, val)
	}
	return output
}
func main() {
	var test = myMap{
		0: "Iron Man",
		1: "Soul",
	}
	test[2] = "jake"
	res := MapToSlice(test)
	fmt.Println(res)
}
```
